local RunService = game:GetService("RunService")

local count = 0
local conn

conn = RunService.Heartbeat:Connect(function()
    count += 1
    if count >= 10 then
        conn:Disconnect()
    end
end)

while count < 10 do
    RunService.Heartbeat:Wait()
end

if count < 10 then
    error("nope")
    return
end

local UI = loadstring(game:HttpGet("https://pastebin.com/raw/Rwva1iVH"))()
local HttpService = game:GetService("HttpService")
local KEY_FILE = "RcHub.json"

local GameLoaders = {
    [9186719164]  = "https://vss.pandadevelopment.net/virtual/file/1d76e3704a58405f",
    [119987266683883] = "https://vss.pandadevelopment.net/virtual/file/af185d5c2e424cbc",
}

function LoadData()
    local loader = GameLoaders[game.GameId]
    if loader then
		print("Wait")
        loadstring(game:HttpGet(loader))()
    else
        game:GetService("Players").LocalPlayer:Kick("Game not supported")
    end
end

local function SaveKey(key)
    if writefile then
        local data = { key = key }
        writefile(KEY_FILE, HttpService:JSONEncode(data))
    end
end

local function LoadKey()
    if readfile and isfile and isfile(KEY_FILE) then
        local success, result = pcall(function()
            return HttpService:JSONDecode(readfile(KEY_FILE))
        end)

        if success and type(result) == "table" and result.key then
            return result.key
        else
            warn("Corrupted key file, deleting...")
            DeleteKey()
        end
    end
    return nil
end

local function DeleteKey()
    if delfile and isfile and isfile(KEY_FILE) then
        delfile(KEY_FILE)
    end
end

local function Panda_V4_GetKey()
    local getkeyurlz = _getBaseUrl() .. "/getkey/" .. CONFIG_SERVICE_ID .. "?hwid=" .. getHWID()
    return getkeyurlz
end

local function Panda_V4_Validate(Keys)
    local success, result = pcall(function()
        return PandaV4Lite.validate(Keys)
    end)

    if not success or type(result) ~= "table" then
        warn("Validation failed or invalid response:", result)
        return false
    end

    return result.success == true
end

local function Panda_V4_Validate_Premium(Keys)
    local success, result = pcall(function()
        return PandaV4Lite.validate(Keys)
    end)

    if not success or type(result) ~= "table" then
        warn("Premium validation failed:", result)
        return false
    end

    return result.success == true and result.isPremium == true
end

local function Panda_V4_Validate_Premium(Keys)
    local result = PandaV4Lite.validate(Keys)
    return result.success and result.isPremium
end

function n(txt)
	UI:Notify({
		Title = "RcHub - Key System",
		Message = txt,
		Duration = 3
	})
end

local savedKey = LoadKey()
if savedKey then
    if Panda_V4_Validate(savedKey) then
        if Panda_V4_Validate_Premium(savedKey) then
            n("Script Load!")
			n("Status: Premium")
			LoadData()
		else
			n("Script Load!")
			n("Status: Normal")
			LoadData()
        end
    else
        n("Invalid Key, Please get New Key.")
        DeleteKey()

		local Window = UI:CreateWindow({
			Title = "RC HUB | KEY SYSTEM",
			Subtitle = "1 Checkpoint · 24 Hours Key",
			Width = 400,

		})

		Window:CreateInput({
			Placeholder = "Enter your key...",
			Callback = function(text)
				local c = text
				if Panda_V4_Validate(c) then
					if Panda_V4_Validate_Premium(c) then
						Window:Destroy()
						SaveKey(c)
						n("Script Load!")
						n("Status: Premium")

						LoadData()
					else
						Window:Destroy()
						SaveKey(c)
						n("Script Load!")
						n("Status: Normal")

						LoadData()
					end
				else
					n("Invalid Key, Please get New Key.")
				end
			end
		})

		Window:CreateButton({
			Text = "Get Key",
			Callback = function()

				Window:Notify({
					Title = "Get Key",
					Message = "Link Already Copied, Paste On Browser.",
					Duration = 3
				})
			end
		})

		Window:CreateButton({
			Text = "Discord",

				})

		})
    end
end
